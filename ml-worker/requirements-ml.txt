# =============================================================================
# REAL-MODEL dependencies for the AETHER ml-worker (the "make it real" stack).
# NOT installed by default (they are large, ~5-7 GB with CUDA). Enable via:
#
#     docker compose build --build-arg INSTALL_ML=true ml-worker
#     # or, local dev:  pip install -r requirements-ml.txt
#
# Powers: CodeBERT text embeddings, ResNet-50 image embeddings, CLIP stego
# detection, FAISS/UMAP/KMeans clustering, SHAP/LIME explainability, and the
# auto-downloaded HuggingFace malware/phishing classifier.
# -----------------------------------------------------------------------------
# CROSS-PLATFORM torch: the DEFAULT PyPI wheel already resolves correctly per
# platform — CUDA 12.x build on x86_64 Linux (uses the GPU), CPU build on
# aarch64 Linux (Apple-Silicon Docker, AWS Graviton, etc.). So we do NOT pin a
# CUDA index here; that keeps the image buildable everywhere.
#   * Force a specific CUDA:  pip install --extra-index-url https://download.pytorch.org/whl/cu121 ...
#   * Force CPU-only (smaller): pip install --extra-index-url https://download.pytorch.org/whl/cpu ...
# ML_DEVICE=auto then uses the GPU when present and falls back to CPU otherwise.
# torch>=2.6 is REQUIRED: newer transformers refuse to load pickle (.bin) weights
# (e.g. CodeBERT) on torch<2.6 for security — older torch silently falls back to mock.
# =============================================================================
torch==2.6.0
torchvision==0.21.0

transformers>=4.44,<5
open_clip_torch>=2.24

# Similarity clustering. faiss-gpu is discontinued on pip (PyPI size limit); the
# reference corpus is tiny so faiss-cpu is the correct, portable choice.
faiss-cpu>=1.8
umap-learn>=0.5.6
scikit-learn>=1.4

# Explainability.
shap>=0.46
lime>=0.2.0.1

# Imaging + numerics. numpy pinned <2.1 to avoid ABI breaks with the faiss/shap/
# umap wheels in this window.
pillow>=10.3
numpy>=1.26,<2.1
scipy>=1.11
